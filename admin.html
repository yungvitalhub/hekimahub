📋 Supabase Table Schema

Run this SQL in your Supabase SQL Editor to create the ebooks table:

create table if not exists ebooks (
  id          uuid default gen_random_uuid() primary key,
  title       text not null,
  author      text not null,
  category    text,
  description text,
  price       numeric(10,2) default 0,
  cover_url   text,
  pdf_url     text,
  buy_url     text,
  tags        text[],
  is_published boolean default false,
  created_at  timestamptz default now()
);

-- Enable Row Level Security
alter table ebooks enable row level security;

-- Allow anyone to READ published ebooks (for your main site)
create policy "Public can view published"
  on ebooks for select
  using (is_published = true);

-- Allow authenticated users full access (for this admin panel)
create policy "Auth users manage all"
  on ebooks for all
  using (auth.role() = 'authenticated');
